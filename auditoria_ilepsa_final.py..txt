# PASO 2: PREPROCESAMIENTO - EXTRACCIÓN DE DATASET
# (A partir de los videos originales de planta)
# ------------------------------------------
def extraer_dataset_de_video(ruta_video, carpeta_destino):
    if not os.path.exists(carpeta_destino):
        os.makedirs(carpeta_destino)
    
    cap = cv2.VideoCapture(ruta_video)
    frame_count = 0
    saved_count = 0
    
    while cap.isOpened():
        ret, frame = cap.read()
        if not ret: break
        # Guardar 1 frame por cada 10 para evitar imágenes casi idénticas
        if frame_count % 10 == 0:
            cv2.imwrite(f"{carpeta_destino}/ilepsa_frame_{saved_count}.jpg", frame)
            saved_count += 1
        frame_count += 1
    cap.release()
    print(f"✅ Extracción terminada: {saved_count} imágenes listas para etiquetar.")

# ------------------------------------------
# PASO 5: MOTOR DE AUDITORÍA HÍBRIDO (IA + VISIÓN CLÁSICA)
# (Lógica de prioridad: Botella -> Etiqueta -> Llenado)
# ------------------------------------------
def motor_auditoria_ilepsa(frame, modelo_cargado, umbral_transparencia=145):
    # Detección con confianza mínima de 0.65
    results = modelo_cargado.track(frame, persist=True, verbose=False)
    
    botellas = []
    etiquetas = []
    r = results[0]
    
    if r.boxes is not None:
        for box in r.boxes:
            cls = int(box.cls[0])
            coords = box.xyxy[0].tolist()
            if cls == 0: botellas.append(coords)
            elif cls == 1: etiquetas.append(coords)

    res_frame = {"OK": 0, "SIN_ETIQUETA": 0, "LLENADO_BAJO": 0}

    for b in botellas:
        bx1, by1, bx2, by2 = b
        # 1. Verificar Etiqueta (Intersección)
        tiene_etiqueta = any(bx1 < (e[0]+e[2])/2 < bx2 and by1 < (e[1]+e[3])/2 < by2 for e in etiquetas)

        if not tiene_etiqueta:
            res_frame["SIN_ETIQUETA"] += 1
        else:
            # 2. Verificar Llenado (ROI en cuello 15%-35%)
            alto = by2 - by1
            roi = frame[int(by1+alto*0.15):int(by1+alto*0.35), int(bx1):int(bx2)]
            if roi.size > 0:
                gray_roi = cv2.cvtColor(roi, cv2.COLOR_BGR2GRAY)
                suave = cv2.GaussianBlur(gray_roi, (5, 5), 0)
                if np.mean(suave) > umbral_transparencia:
                    res_frame["LLENADO_BAJO"] += 1
                else:
                    res_frame["OK"] += 1
            else:
                res_frame["OK"] += 1
    return res_frame

# ------------------------------------------
# PASO 6: VALIDACIÓN FINAL (EL DATASET DE 55 IMÁGENES)
# ------------------------------------------
ruta_prueba = '/content/drive/MyDrive/Proyecto_ILEPSA/dataset/images'

archivos = [f for f in os.listdir(ruta_prueba) if f.lower().endswith(('.png', '.jpg'))]
stats_totales = {"OK": 0, "SE": 0, "LL": 0}

for img_name in archivos:
    img = cv2.imread(os.path.join(ruta_prueba, img_name))
    reporte = motor_auditoria_ilepsa(img, modelo_final)
    stats_totales["OK"] += reporte["OK"]
    stats_totales["SE"] += reporte["SIN_ETIQUETA"]
    stats_totales["LL"] += reporte["LLENADO_BAJO"]

# ------------------------------------------
# PASO 7: GENERACIÓN DE EVIDENCIA ESTADÍSTICA

labels = ['Calidad OK', 'Sin Etiqueta', 'Nivel Bajo']
